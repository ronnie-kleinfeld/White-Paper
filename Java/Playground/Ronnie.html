מחלקה Box

public class _Box {
    public String color;
    public int width;
    public int height;
    public int length;

    public _Box() {
    }
    public _Box(String color, int width, int height, int length) {
        this.color = color;
        this.width = width;
        this.height = height;
        this.length = length;
    }

    public int area() {
        return width * height * length;
    }

    public String ToString() {
        return "Box, Color:" + color + " width:" + width + " height:" + height + " length:" + length;
    }
}

מחיקת איבר במערך

public static int[] RemoveByIndex(int[] arr, int removeIndex) {
    int[] newArr = new int[arr.length - 1];
    int newIndex = 0;

    for (int i = 0; i < arr.length; i++) {
        if (i != removeIndex) {
            newArr[newIndex++] = arr[i];
        }
    }
    return newArr;
}

    }

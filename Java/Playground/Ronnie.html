מחלקה BOX

public class Box {
    public String color;
    public int width;
    public int height;

    public Box() {
    }

    public Box(String color, int width, int height) {
        this.color = color;
        this.width = width;
        this.height = height;
    }

    public int Area() {
        return width * height;
    }

    public String ToString() {
        return "Rectangle, Color:" + color + " width:" + width + " height:" + height;
    }
}

מחיקת איבר במערך

public static int[] RemoveByIndex(int[] arr, int removeIndex) {
    int[] newArr = new int[arr.length - 1];
    int newIndex = 0;

    for (int i = 0; i < arr.length; i++) {
        if (i !=removeIndex) {
            newArr[newIndex++]=arr[i];
        }
    }
    return newArr;
}

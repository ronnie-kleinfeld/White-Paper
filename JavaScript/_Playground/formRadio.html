function getRadioValue(radioArray) {
    var i
    for (i = 0; i < radioArray.length; i++) {
        if (radioArray[i].checked) return radioArray[i].value
    }
    return ""
}

function showSelectedRadioValue(formReference) {
    var rvalue = getRadioValue(formRef.Radio1)
    if (rvalue == "") {
        alert("None of the radio buttons are checked.\n",
            "After you check one you will never see this message again.\n",
            "Use the CHECKED attribtue in the input tag to",
            "make sure one button is always selected.")
    } else {
        alert("You chose: " + rvalue)
    }
}
